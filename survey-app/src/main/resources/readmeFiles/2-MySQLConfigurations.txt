## This file gives you the commands to run in mysql command line.
## If you can use mysql command, also you can connect to mysql server by command line.

/*
    Keywords will be used in commands are:
    'username' : username of your mysql (I will use root as default)
    'password' : password you will use
    'database_name' : the name for our database
    'user_name' : the name of user will be used from everywhere
    'host_name' : host_name or host_address to use which we will use
    'type_of_permission' : the type of MySQL permissions
                            CREATE, SELECT, UPDATE, DELETE, DROP, INSERT, ALL PRIVILEGES, GRANT OPTION
    'table_name' : the name of tables to give this permission
                            (For all of tables use '*' like 'database_name'.*)

    ** First we have to connect our database:

        mysql -u 'username' -p
        'password'

    ** Then we have to create database:

        create database 'database_name';

    ** Create user:

        create user 'user_name'@'host_name' identified by 'password';

    ** Grant user permissions:

        grant 'type_of_permission' ON 'database_name'.'table_name' TO 'user_name'@'host_name';

*/
